\subsection*{Teil A â€“ Funktion einer exponentialverteilten Zufallsvariablen}
% ----------------------------------------------------------------

Die Zufallsvariable $X$ ist exponentialverteilt
(Abschnitt~\ref{sec:exponentialverteilung}) mit Parameter $\lambda > 0$:

$$
p_X(x) = \begin{cases} \lambda\, e^{-\lambda x} & , \quad x \geq 0 \\[2pt]
0 & , \quad \text{sonst.} \end{cases}
$$

Der TrĂ¤ger von $X$ ist also $[0, \infty)$. Diese EinschrĂ¤nkung ist fĂĽr die
Wahl der Umkehrfunktion entscheidend.

% ---- A1 --------------------------------------------------------
\subsubsection*{A1 â€“ Quadratische Transformation $Y = X^2$}

\begin{beispiel}[Funktion einer Zufallsvariablen -- $Y = X^2$]

\textbf{Gegeben:} Exponentialverteilte Zufallsvariable $X$ mit Dichte
$p_X(x) = \lambda e^{-\lambda x}$ fĂĽr $x \geq 0$ und Parameter $\lambda > 0$.
Transformation $Y = g(X) = X^2$.

\textbf{Gesucht:} Die Wahrscheinlichkeitsdichte $p_Y(y)$ der transformierten
Zufallsvariablen $Y$.

\textbf{Formel} (Transformationssatz fĂĽr streng monotone Funktionen,
Abschnitt~\ref{sec:monotone-transformation}):

$$
p_Y(y) = \frac{p_X\!\left(g^{-1}(y)\right)}{\left|\, g'\!\left(g^{-1}(y)\right)\right|}
$$

\textbf{Rechnung:}

Zwar ist $g(x) = x^2$ auf ganz $\mathbb{R}$ \emph{nicht} streng monoton, auf dem
TrĂ¤ger $x \geq 0$ von $X$ jedoch streng monoton steigend. Es kommt daher nur die
\emph{positive} Wurzel als Umkehrfunktion in Frage:

$$
x = g^{-1}(y) = \sqrt{y}, \qquad y \geq 0.
$$

Die Ableitung von $g(x) = x^2$ ist $g'(x) = 2x$, an der Stelle $x = \sqrt{y}$
also

$$
\left|\, g'\!\left(\sqrt{y}\right)\right| = 2\sqrt{y}.
$$

Einsetzen in den Transformationssatz liefert fĂĽr $y \geq 0$:

$$
p_Y(y) = \frac{p_X\!\left(\sqrt{y}\right)}{2\sqrt{y}}
       = \frac{\lambda\, e^{-\lambda \sqrt{y}}}{2\sqrt{y}}.
$$

FĂĽr $y < 0$ existiert kein reelles Urbild, also $p_Y(y) = 0$.

\textbf{LĂ¶sung:}

$$
\boxed{\;p_Y(y) = \begin{cases}
\dfrac{\lambda}{2\sqrt{y}}\, e^{-\lambda \sqrt{y}} & , \quad y > 0 \\[8pt]
0 & , \quad \text{sonst.}
\end{cases}\;}
$$

An der Stelle $y \to 0^+$ besitzt die Dichte wegen des Faktors
$1/(2\sqrt{y})$ eine (integrierbare) Polstelle -- die Wahrscheinlichkeitsmasse
wird durch das Quadrieren nahe null stark â€žgestaucht``. Zur Kontrolle gilt
weiterhin $\int_0^\infty p_Y(y)\,\mathrm{d}y = 1$ (Substitution
$y = x^2$ fĂĽhrt auf das Integral der Ausgangsdichte).

\end{beispiel}

% ---- A2 --------------------------------------------------------
\subsubsection*{A2 â€“ Lineare Transformation $Y = aX + b$}

\begin{beispiel}[Funktion einer Zufallsvariablen -- $Y = aX + b$]

\textbf{Gegeben:} Exponentialverteilte Zufallsvariable $X$ mit Dichte
$p_X(x) = \lambda e^{-\lambda x}$ fĂĽr $x \geq 0$. Lineare Transformation
$Y = g(X) = aX + b$ mit Konstanten $a \neq 0$ und $b \in \mathbb{R}$.

\textbf{Gesucht:} Die Wahrscheinlichkeitsdichte $p_Y(y)$ der transformierten
Zufallsvariablen $Y$.

\textbf{Formel} (Transformationssatz fĂĽr streng monotone Funktionen,
Abschnitt~\ref{sec:monotone-transformation}):

$$
p_Y(y) = \frac{p_X\!\left(g^{-1}(y)\right)}{\left|\, g'\!\left(g^{-1}(y)\right)\right|}
$$

\textbf{Rechnung:}

Die lineare Funktion $g(x) = ax + b$ ist wegen $a \neq 0$ streng monoton und
eindeutig umkehrbar:

$$
x = g^{-1}(y) = \frac{y - b}{a}, \qquad g'(x) = a, \qquad |g'(x)| = |a|.
$$

Einsetzen ergibt zunĂ¤chst formal

$$
p_Y(y) = \frac{1}{|a|}\, p_X\!\left(\frac{y - b}{a}\right).
$$

Die Ausgangsdichte $p_X$ ist nur fĂĽr ein nichtnegatives Argument von null
verschieden, also fĂĽr $\dfrac{y - b}{a} \geq 0$. FĂĽr den typischen Fall
$a > 0$ bedeutet das $y \geq b$, und man erhĂ¤lt

$$
p_Y(y) = \frac{\lambda}{a}\,
         \exp\!\left(-\lambda\,\frac{y - b}{a}\right), \qquad y \geq b.
$$

\textbf{LĂ¶sung} (Fall $a > 0$):

$$
\boxed{\;p_Y(y) = \begin{cases}
\dfrac{\lambda}{a}\, \exp\!\left(-\dfrac{\lambda}{a}\,(y - b)\right)
  & , \quad y \geq b \\[8pt]
0 & , \quad y < b.
\end{cases}\;}
$$

$Y$ ist damit wieder \emph{exponentialverteilt}, allerdings um $b$ nach rechts
verschoben und mit dem neuen Parameter $\lambda' = \lambda/a$. Der Faktor $a$
skaliert die Zufallsvariable und verkleinert entsprechend den Ratenparameter
(im Fall $a < 0$ kehrt sich der TrĂ¤ger zu $y \leq b$ um; die kompakte
Bedingung lautet allgemein $\tfrac{y-b}{a} \geq 0$, und der Vorfaktor ist
$\lambda/|a|$).

\end{beispiel}

\subsubsection*{B1 â€“ Transformation auf die Standardnormalverteilung}

\begin{beispiel}[Standardisierung $X \sim \mathcal{N}(\mu_X, \sigma_X^2) \to Y \sim \mathcal{N}(0,1)$]

\textbf{Gegeben:} Normalverteilte Zufallsvariable $X$
(Abschnitt~\ref{sec:normalverteilung}) mit Mittelwert $\mu_X$ und Varianz
$\sigma_X^2 > 0$:

$$
p_X(x) = \frac{1}{\sqrt{2\pi}\,\sigma_X}\,
         \exp\!\left(-\frac{(x - \mu_X)^2}{2\sigma_X^2}\right).
$$

\textbf{Gesucht:} Eine lineare Transformation $Y = g(X)$, so dass $Y$
standardnormalverteilt ist ($\mu_Y = 0$, $\sigma_Y^2 = 1$), sowie der Nachweis
ĂĽber die transformierte Dichte $p_Y(y)$.

\textbf{Formel} (Standardisierung sowie Transformationssatz,
Abschnitt~\ref{sec:monotone-transformation}):

$$
Y = \frac{X - \mu_X}{\sigma_X}, \qquad
p_Y(y) = \frac{p_X\!\left(g^{-1}(y)\right)}{\left|\, g'\!\left(g^{-1}(y)\right)\right|}
$$

\textbf{Rechnung:}

\emph{Ansatz.} Die Standardisierung subtrahiert den Mittelwert und normiert auf
die Standardabweichung:

$$
Y = g(X) = \frac{X - \mu_X}{\sigma_X}.
$$

Dass diese Wahl die geforderten Momente erzeugt, folgt direkt aus der
LinearitĂ¤t von Erwartungswert (Abschnitt~\ref{sec:erwartungswert}) und Varianz:

$$
\mu_Y = \mathrm{E}\{Y\} = \frac{\mathrm{E}\{X\} - \mu_X}{\sigma_X} = 0,
\qquad
\sigma_Y^2 = \mathrm{var}\{Y\} = \frac{\mathrm{var}\{X\}}{\sigma_X^2}
           = \frac{\sigma_X^2}{\sigma_X^2} = 1.
$$

\emph{Dichte.} Die Funktion $g(x) = (x - \mu_X)/\sigma_X$ ist streng monoton
steigend mit Umkehrfunktion und Ableitung

$$
x = g^{-1}(y) = \sigma_X\, y + \mu_X, \qquad
g'(x) = \frac{1}{\sigma_X}, \qquad |g'(x)| = \frac{1}{\sigma_X}.
$$

Einsetzen in den Transformationssatz:

$$
p_Y(y) = \frac{p_X\!\left(\sigma_X\, y + \mu_X\right)}{1/\sigma_X}
       = \sigma_X\, p_X\!\left(\sigma_X\, y + \mu_X\right).
$$

Nun wird $x = \sigma_X y + \mu_X$ in die Normaldichte eingesetzt; im Exponenten
gilt $(x - \mu_X)^2 = (\sigma_X y)^2 = \sigma_X^2\, y^2$:

$$
p_Y(y) = \sigma_X \cdot \frac{1}{\sqrt{2\pi}\,\sigma_X}\,
         \exp\!\left(-\frac{\sigma_X^2\, y^2}{2\sigma_X^2}\right)
       = \frac{1}{\sqrt{2\pi}}\, \exp\!\left(-\frac{y^2}{2}\right).
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;Y = \frac{X - \mu_X}{\sigma_X}
\quad\Longrightarrow\quad
p_Y(y) = \frac{1}{\sqrt{2\pi}}\, e^{-y^2/2}
= \mathcal{N}(y;\, 0,\, 1)\;}
$$

$Y$ ist also standardnormalverteilt. Die Standardabweichung $\sigma_X$ aus dem
Betrag der Ableitung kĂĽrzt sich exakt gegen den Vorfaktor der Ausgangsdichte,
und die Verschiebung um $\mu_X$ verschwindet im Exponenten. Diese
Standardisierung ist die Grundlage fĂĽr die Nutzung tabellierter Werte der
Standardnormalverteilung $\Phi(y)$ fĂĽr beliebige normalverteilte GrĂ¶Ăźen.

\end{beispiel}

\subsection*{Aufgabe 10.1 â€“ Transformation auf eine Standard-Gleichverteilung}
% ----------------------------------------------------------------

\begin{beispiel}[Gleichverteilung auf einem Intervall zur Standard-Gleichverteilung]

\textbf{Gegeben:} Eine auf dem Intervall $[a, b]$ (mit $a < b$) gleichverteilte
Zufallsvariable $X$ (Abschnitt~\ref{sec:gleichverteilung}):

$$
p_X(x) = \begin{cases} \dfrac{1}{b - a} & , \quad a \leq x \leq b \\[6pt]
0 & , \quad \text{sonst.} \end{cases}
$$

\textbf{Gesucht:} Eine Transformation $Y = g(X)$, so dass $Y$ auf dem Intervall
$\left[-\tfrac{1}{2}, \tfrac{1}{2}\right]$ gleichverteilt ist, sowie der
Nachweis ĂĽber die transformierte Dichte $p_Y(y)$.

\textbf{Formel} (lineare Transformation $Y = cX + d$ und Transformationssatz,
Abschnitt~\ref{sec:monotone-transformation}):

$$
p_Y(y) = \frac{p_X\!\left(g^{-1}(y)\right)}{\left|\, g'\!\left(g^{-1}(y)\right)\right|}
$$

\textbf{Rechnung:}

\emph{Ansatz.} Da beide Intervalle die Endpunkte einer Gleichverteilung
festlegen, genĂĽgt eine \emph{lineare} und streng monoton steigende
Transformation $Y = cX + d$, welche die Intervallgrenzen aufeinander abbildet:

$$
g(a) = -\tfrac{1}{2}, \qquad g(b) = +\tfrac{1}{2}.
$$

Die Steigung ergibt sich aus dem VerhĂ¤ltnis der IntervalllĂ¤ngen:

$$
c = \frac{\tfrac{1}{2} - \left(-\tfrac{1}{2}\right)}{b - a} = \frac{1}{b - a}.
$$

Der Versatz $d$ folgt aus $g(a) = -\tfrac{1}{2}$:

$$
d = -\tfrac{1}{2} - \frac{a}{b - a} = -\frac{a + b}{2\,(b - a)}.
$$

Zusammengefasst lĂ¤sst sich die Transformation kompakt als Zentrierung um den
Mittelwert $\mu_X = \tfrac{a+b}{2}$ und Normierung auf die IntervalllĂ¤nge
schreiben:

$$
Y = g(X) = \frac{X - \dfrac{a + b}{2}}{b - a}.
$$

\emph{Kontrolle der Grenzen:}

$$
g(a) = \frac{a - \tfrac{a+b}{2}}{b - a} = \frac{-\tfrac{b-a}{2}}{b-a} = -\tfrac{1}{2},
\qquad
g(b) = \frac{b - \tfrac{a+b}{2}}{b - a} = \frac{\tfrac{b-a}{2}}{b-a} = +\tfrac{1}{2}.
\quad\checkmark
$$

\emph{Dichte.} Die Funktion $g$ ist streng monoton steigend mit

$$
x = g^{-1}(y) = (b - a)\, y + \frac{a + b}{2}, \qquad
g'(x) = \frac{1}{b - a}, \qquad |g'(x)| = \frac{1}{b - a}.
$$

Einsetzen in den Transformationssatz fĂĽr $y \in \left[-\tfrac12, \tfrac12\right]$
(dort ist $g^{-1}(y) \in [a, b]$, also $p_X = \tfrac{1}{b-a}$):

$$
p_Y(y) = \frac{p_X\!\left(g^{-1}(y)\right)}{|g'|}
       = \frac{\dfrac{1}{b - a}}{\dfrac{1}{b - a}}
       = 1.
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;Y = \frac{X - \dfrac{a + b}{2}}{b - a}
\quad\Longrightarrow\quad
p_Y(y) = \begin{cases} 1 & , \quad -\tfrac{1}{2} \leq y \leq \tfrac{1}{2} \\[4pt]
0 & , \quad \text{sonst.} \end{cases}\;}
$$

$Y$ ist also auf $\left[-\tfrac{1}{2}, \tfrac{1}{2}\right]$ gleichverteilt; die
Dichte hat dort den Wert $1$, da das Zielintervall die LĂ¤nge $1$ besitzt. Die
Transformation entspricht einer Standardisierung der Gleichverteilung: Sie
verschiebt den Mittelwert nach $0$ und skaliert die Spannweite auf $1$
(analog zur Standardisierung der Normalverteilung in Teil~B, dort jedoch mit
$\sigma_X$ statt der IntervalllĂ¤nge $b - a$).

\end{beispiel}

\subsection*{Aufgabe 10.2 â€“ Funktion einer exponentialverteilten Zufallsvariablen}
% ----------------------------------------------------------------

Die Zufallsvariable $X$ ist exponentialverteilt
(Abschnitt~\ref{sec:exponentialverteilung}) mit Parameter $\lambda > 0$,
$p_X(x) = \lambda e^{-\lambda x}$ fĂĽr $x \geq 0$. Da der TrĂ¤ger von $X$
nichtnegativ ist, gilt $|X| = X$; die BetrĂ¤ge in den Teilaufgaben b) und c)

% ---- a) --------------------------------------------------------
\subsubsection*{a) Wurzeltransformation $Y = \sqrt{X}$}

\begin{beispiel}[Funktion einer Zufallsvariablen -- $Y = \sqrt{X}$]

\textbf{Gegeben:} $p_X(x) = \lambda e^{-\lambda x}$ fĂĽr $x \geq 0$,
Transformation $Y = g(X) = \sqrt{X}$.

\textbf{Gesucht:} Die Dichte $p_Y(y)$.

\textbf{Formel} (Transformationssatz fĂĽr streng monotone Funktionen,
Abschnitt~\ref{sec:monotone-transformation}):

$$
p_Y(y) = \frac{p_X\!\left(g^{-1}(y)\right)}{\left|\, g'\!\left(g^{-1}(y)\right)\right|}
$$

\textbf{Rechnung:}

Auf dem TrĂ¤ger $x \geq 0$ ist $g(x) = \sqrt{x}$ streng monoton steigend mit

$$
x = g^{-1}(y) = y^2, \qquad g'(x) = \frac{1}{2\sqrt{x}},
\qquad \left|g'(y^2)\right| = \frac{1}{2\sqrt{y^2}} = \frac{1}{2y}, \quad y > 0.
$$

Einsetzen fĂĽr $y \geq 0$:

$$
p_Y(y) = \frac{p_X(y^2)}{1/(2y)} = 2y\,\lambda\, e^{-\lambda y^2}.
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;p_Y(y) = \begin{cases}
2\lambda\, y\, e^{-\lambda y^2} & , \quad y \geq 0 \\[6pt]
0 & , \quad y < 0.
\end{cases}\;}
$$

$Y$ ist \emph{Rayleigh-verteilt}: Das Wurzelziehen streckt kleine Werte und
erzeugt am Ursprung eine linear ansteigende Dichte statt des Sprungs der
Exponentialverteilung.

\end{beispiel}

% ---- b) --------------------------------------------------------
\subsubsection*{b) Logarithmische Transformation $Y = \ln(|X|)$}

\begin{beispiel}[Funktion einer Zufallsvariablen -- $Y = \ln(|X|)$]

\textbf{Gegeben:} $p_X(x) = \lambda e^{-\lambda x}$ fĂĽr $x \geq 0$,
Transformation $Y = \ln(|X|) = \ln(X)$.

\textbf{Gesucht:} Die Dichte $p_Y(y)$.

\textbf{Formel} (Transformationssatz, Abschnitt~\ref{sec:monotone-transformation}):
$p_Y(y) = p_X(g^{-1}(y)) / |g'(g^{-1}(y))|$.

\textbf{Rechnung:}

FĂĽr $x > 0$ ist $g(x) = \ln x$ streng monoton steigend mit

$$
x = g^{-1}(y) = e^{y}, \qquad g'(x) = \frac{1}{x},
\qquad \left|g'(e^y)\right| = e^{-y}, \quad y \in \mathbb{R}.
$$

Da $g^{-1}(y) = e^y > 0$ fĂĽr alle reellen $y$, liegt das Urbild stets im
TrĂ¤ger, und man erhĂ¤lt fĂĽr alle $y \in \mathbb{R}$:

$$
p_Y(y) = \frac{p_X(e^y)}{e^{-y}} = e^{y}\,\lambda\, e^{-\lambda e^{y}}.
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;p_Y(y) = \lambda\, e^{\,y}\, \exp\!\left(-\lambda\, e^{\,y}\right),
\qquad y \in \mathbb{R}.\;}
$$

Der Logarithmus dehnt den TrĂ¤ger auf ganz $\mathbb{R}$; die Dichte besitzt ein
Maximum bei $y = -\ln\lambda$ und fĂ¤llt fĂĽr groĂźe $y$ doppelt-exponentiell ab
(Verteilung vom Gumbel-Typ).

\end{beispiel}

\subsubsection*{c) Pegeltransformation $Y = 20\,\log_{10}(|X|)$}

\begin{beispiel}[Funktion einer Zufallsvariablen -- $Y = 20\log_{10}(|X|)$]

\textbf{Gegeben:} $p_X(x) = \lambda e^{-\lambda x}$ fĂĽr $x \geq 0$,
Transformation $Y = 20\,\log_{10}(|X|) = 20\,\log_{10}(X)$ (Umrechnung in dB).

\textbf{Gesucht:} Die Dichte $p_Y(y)$.

\textbf{Formel} (Transformationssatz, Abschnitt~\ref{sec:monotone-transformation}).

\textbf{Rechnung:}

$x > 0$ streng monoton steigend. Aus $y = 20\log_{10}x$ folgt

$$

g'(x) = \frac{20}{x\,\ln 10}, \qquad
\left|g'\!\left(10^{y/20}\right)\right| = \frac{20}{10^{\,y/20}\,\ln 10}.

Einsetzen fĂĽr alle $y \in \mathbb{R}$ (auch hier ist $10^{y/20} > 0$ stets im

$$

       = \frac{\ln 10}{20}\,\lambda\, 10^{\,y/20}\,
         \exp\!\left(-\lambda\, 10^{\,y/20}\right).
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;p_Y(y) = \frac{\ln 10}{20}\,\lambda\, 10^{\,y/20}\,
\exp\!\left(-\lambda\, 10^{\,y/20}\right), \qquad y \in \mathbb{R}.\;}
$$

Die dB-Kennlinie ist bis auf den konstanten Faktor $20/\ln 10$ dieselbe
logarithmische Stauchung wie in b); entsprechend hat $p_Y$ dieselbe
doppelt-exponentielle Form, nur horizontal skaliert.

\end{beispiel}

\subsection*{Aufgabe 10.3 â€“ Funktion einer gleichverteilten Zufallsvariablen}
% ----------------------------------------------------------------

\begin{beispiel}[Funktion einer Zufallsvariablen -- $Y = -\tfrac{1}{\alpha}\ln(X)$]

\textbf{Gegeben:} $X$ ist auf $(0,1)$ gleichverteilt
(Abschnitt~\ref{sec:gleichverteilung}), also $p_X(x) = 1$ fĂĽr $0 < x < 1$
(sonst $0$). Transformation $Y = g(X) = -\dfrac{1}{\alpha}\ln(X)$ mit
$\alpha > 0$.

\textbf{Gesucht:} Erwartungswert $\mathrm{E}\{Y\}$, Varianz $\mathrm{var}\{Y\}$
und Dichte $p_Y(y)$; welche bekannte Verteilung liegt fĂĽr $\alpha = \tfrac12$
vor?

\textbf{Formel} (Transformationssatz, Abschnitt~\ref{sec:monotone-transformation};
Momente, Abschnitt~\ref{sec:momente}).

\textbf{Rechnung:}

\emph{Dichte.} FĂĽr $x \in (0,1)$ ist $\ln x < 0$, also $y > 0$; $g$ ist streng
monoton \emph{fallend} mit

$$
x = g^{-1}(y) = e^{-\alpha y}, \qquad
g'(x) = -\frac{1}{\alpha x}, \qquad
\left|g'\!\left(e^{-\alpha y}\right)\right| = \frac{1}{\alpha\, e^{-\alpha y}}
= \frac{e^{\alpha y}}{\alpha}.
$$

FĂĽr $y > 0$ liegt $e^{-\alpha y} \in (0,1)$ im TrĂ¤ger, dort $p_X = 1$:

$$
p_Y(y) = \frac{p_X\!\left(e^{-\alpha y}\right)}{e^{\alpha y}/\alpha}
       = \alpha\, e^{-\alpha y}.
$$

$Y$ ist also \emph{exponentialverteilt} mit Parameter $\alpha$
(Abschnitt~\ref{sec:exponentialverteilung}).

\emph{Momente.} Aus der Exponentialverteilung folgt unmittelbar

$$
\mathrm{E}\{Y\} = \int_0^\infty y\,\alpha e^{-\alpha y}\,\mathrm{d}y
= \frac{1}{\alpha}, \qquad
\mathrm{E}\{Y^2\} = \int_0^\infty y^2\,\alpha e^{-\alpha y}\,\mathrm{d}y
= \frac{2}{\alpha^2},
$$

$$
\mathrm{var}\{Y\} = \mathrm{E}\{Y^2\} - \left(\mathrm{E}\{Y\}\right)^2
= \frac{2}{\alpha^2} - \frac{1}{\alpha^2} = \frac{1}{\alpha^2}.
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;p_Y(y) = \begin{cases} \alpha\, e^{-\alpha y} & , \; y \geq 0 \\[4pt]
0 & , \; y < 0 \end{cases}
\qquad
\mathrm{E}\{Y\} = \frac{1}{\alpha}, \qquad
\mathrm{var}\{Y\} = \frac{1}{\alpha^2}.\;}
$$

FĂĽr $\alpha = \tfrac12$ wird $p_Y(y) = \tfrac12 e^{-y/2}$ ($y \geq 0$) -- das ist
zugleich die Dichte der \emph{Chi-Quadrat-Verteilung mit zwei Freiheitsgraden}
($\chi^2_2$). Dieses Verfahren (Invertieren der Verteilungsfunktion,
Abschnitt~\ref{sec:verteilungsfunktion}) ist der \emph{Inversionsansatz} zur
Erzeugung exponentialverteilter Zufallszahlen aus gleichverteilten.

\end{beispiel}

\subsection*{Aufgabe 10.6 â€“ Faltung zweier Normalverteilungen}

\begin{beispiel}[Dichte der Summe zweier unabhĂ¤ngiger $\mathcal{N}(0,\sigma^2)$]

= \frac{1}{2\sigma\sqrt{\pi}}\, e^{-z^2/(4\sigma^2)}.
$$

\textbf{LĂ¶sung:}

$$

($\sigma_Z^2 = 2\sigma^2$, $\sigma_Z = \sqrt{2}\,\sigma$). Die Faltung zweier

Faltung stabil.

\end{beispiel}

\subsection*{Aufgabe 10.7 â€“ Buffonsches Nadelproblem}

Eine Nadel der LĂ¤nge $2a$ fĂ¤llt auf eine Ebene mit parallelen Linien im Abstand

\textbf{Gegeben:} Nadel $2a$, Linienabstand $2b$, $a < b$; $X$ = Abstand

\subsection*{Aufgabe 10.8 â€“ Funktionen zweier exponentialverteilter Zufallsvariablen}

Gegeben sind zwei statistisch unabhĂ¤ngige
(Abschnitt~\ref{sec:statistische-unabhaengigkeit}) exponentialverteilte
Zufallsvariablen mit
$p_{X_1}(x_1) = \alpha e^{-\alpha x_1}\,\epsilon(x_1)$ und
$p_{X_2}(x_2) = \beta e^{-\beta x_2}\,\epsilon(x_2)$, $\alpha,\beta > 0$.

% ---- a) --------------------------------------------------------
\subsubsection*{a) Gewichtete Summe $Y = 2X_1 + X_2$}

\begin{beispiel}[Funktion zweier Zufallsvariablen -- $Y = 2X_1 + X_2$]

\textbf{Gegeben:} $X_1 \sim \text{Exp}(\alpha)$, $X_2 \sim \text{Exp}(\beta)$,
unabhĂ¤ngig.

\textbf{Gesucht:} Die Dichte $p_Y(y)$ von $Y = 2X_1 + X_2$.

\textbf{Formel} (Skalierung nach Abschnitt~\ref{sec:monotone-transformation} und
Faltung nach Abschnitt~\ref{sec:funktion-zwei-zv}).

\textbf{Rechnung:}

ZunĂ¤chst die Dichte von $U = 2X_1$: mit $u = 2x_1$ ist
$p_U(u) = \tfrac12 p_{X_1}(u/2) = \tfrac{\alpha}{2} e^{-(\alpha/2)u}$ fĂĽr
$u \geq 0$, also $U \sim \text{Exp}(\alpha/2)$. Nun $Y = U + X_2$ als Faltung
zweier Exponentialdichten mit Raten $\alpha' = \alpha/2$ und $\beta$
(Fall $\alpha \neq 2\beta$, d.\,h. $\alpha' \neq \beta$):

$$
p_Y(y) = \int_0^{y} \alpha' e^{-\alpha'\tau}\,\beta e^{-\beta(y-\tau)}\,\mathrm{d}\tau
= \alpha'\beta\, e^{-\beta y}\,\frac{1 - e^{-(\alpha'-\beta)y}}{\alpha'-\beta}
= \frac{\alpha'\beta}{\alpha'-\beta}\left(e^{-\beta y} - e^{-\alpha' y}\right).
$$

Mit $\alpha' = \alpha/2$ und Erweitern um den Faktor $2$:

$$
p_Y(y) = \frac{\alpha\beta}{\alpha - 2\beta}
\left(e^{-\beta y} - e^{-\alpha y/2}\right), \qquad y \geq 0.
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;p_Y(y) = \frac{\alpha\beta}{\alpha - 2\beta}
\left(e^{-\beta y} - e^{-\alpha y/2}\right)\,\epsilon(y)
\quad (\alpha \neq 2\beta).\;}
$$

Im Sonderfall $\alpha = 2\beta$ ($\alpha' = \beta$) liefert das Integral
$p_Y(y) = \beta^2\, y\, e^{-\beta y}\,\epsilon(y)$ (Erlang-/Gamma-Dichte der
Ordnung $2$).

\end{beispiel}

% ---- b) --------------------------------------------------------
\subsubsection*{b) Differenz $Y = X_1 - X_2$}

\begin{beispiel}[Funktion zweier Zufallsvariablen -- $Y = X_1 - X_2$]

\textbf{Gegeben:} $X_1 \sim \text{Exp}(\alpha)$, $X_2 \sim \text{Exp}(\beta)$,
unabhĂ¤ngig.

\textbf{Gesucht:} Die Dichte $p_Y(y)$ von $Y = X_1 - X_2$.

\textbf{Formel} (Faltung fĂĽr die Differenz,
Abschnitt~\ref{sec:funktion-zwei-zv}):
$p_Y(y) = \int_{-\infty}^{\infty} p_{X_1}(x)\, p_{X_2}(x - y)\,\mathrm{d}x$.

\textbf{Rechnung:}

Der Integrand verlangt $x \geq 0$ und $x - y \geq 0$, also
$x \geq \max(0, y)$.

\emph{Fall $y \geq 0$:}

$$
p_Y(y) = \int_{y}^{\infty} \alpha e^{-\alpha x}\,\beta e^{-\beta(x-y)}\,\mathrm{d}x
= \alpha\beta\, e^{\beta y}\int_{y}^{\infty} e^{-(\alpha+\beta)x}\,\mathrm{d}x
= \frac{\alpha\beta}{\alpha+\beta}\, e^{-\alpha y}.
$$

\emph{Fall $y < 0$:} (untere Grenze $x = 0$)

$$
p_Y(y) = \int_{0}^{\infty} \alpha e^{-\alpha x}\,\beta e^{-\beta(x-y)}\,\mathrm{d}x
= \alpha\beta\, e^{\beta y}\int_{0}^{\infty} e^{-(\alpha+\beta)x}\,\mathrm{d}x
= \frac{\alpha\beta}{\alpha+\beta}\, e^{\beta y}.
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;p_Y(y) = \frac{\alpha\beta}{\alpha+\beta}
\begin{cases} e^{-\alpha y} & , \quad y \geq 0 \\[4pt]
e^{\beta y} & , \quad y < 0. \end{cases}\;}
$$

Das ist eine \emph{asymmetrische Laplace-Dichte}
(Abschnitt~\ref{sec:laplaceverteilung}): rechts mit Rate $\alpha$, links mit
Rate $\beta$ abfallend; fĂĽr $\alpha = \beta$ ergibt sich die symmetrische
Laplace-Verteilung.

\end{beispiel}

% ---- c) --------------------------------------------------------
\subsubsection*{c) Maximum $Y = \max(X_1, X_2)$}

\begin{beispiel}[Funktion zweier Zufallsvariablen -- $Y = \max(X_1, X_2)$]

\textbf{Gegeben:} $X_1 \sim \text{Exp}(\alpha)$, $X_2 \sim \text{Exp}(\beta)$,

\textbf{Gesucht:} Die Dichte $p_Y(y)$ von $Y = \max(X_1, X_2)$.

$$

Ableiten liefert die Dichte:

$$

\textbf{LĂ¶sung:}

$$
\boxed{\;p_Y(y) = \left[\alpha e^{-\alpha y} + \beta e^{-\beta y}

verschwindet die Dichte bei $y = 0$ ($p_Y(0) = 0$).

\end{beispiel}

\subsubsection*{d) Minimum $Y = \min(X_1, X_2)$}

\begin{beispiel}[Funktion zweier Zufallsvariablen -- $Y = \min(X_1, X_2)$]

\end{beispiel}

\subsection*{Zusatzaufgabe â€“ Faltung von Rechteckimpuls und Exponentialsystem}

Ein LTI-System besitzt die Impulsantwort

$$

$$

= \begin{cases} 3 & , \quad 2T \leq t < 4T \\[2pt] 0 & , \quad \text{sonst.} \end{cases}
$$

Dabei wurde die Skript-Definition
$\operatorname{rect}(u) = 1$ fĂĽr $0 \leq u < 1$ (sonst $0$) verwendet; das
Argument $\tfrac{t-2T}{2T}$ liegt genau fĂĽr $2T \leq t < 4T$ in diesem Bereich.
Der Rechteckimpuls hat also die Breite $2T$, die HĂ¶he $3$ und beginnt bei $2T$.

% ---- a) --------------------------------------------------------
\subsubsection*{a) BIBO-StabilitĂ¤t des Systems}

\begin{beispiel}[BIBO-StabilitĂ¤t ĂĽber die absolute Integrierbarkeit der Impulsantwort]

\textbf{Gegeben:} Impulsantwort $h(t) = 3\, e^{-t/(4T)}\, \epsilon(t)$ mit
$T > 0$.

\textbf{Gesucht:} Ist das durch $h(t)$ beschriebene LTI-System BIBO-stabil?

\textbf{Formel} (hinreichende und fĂĽr LTI-Systeme notwendige Bedingung --
absolute Integrierbarkeit der Impulsantwort,
Abschnitt~\ref{sec:faltung-eigenschaften}):

$$
\int_{-\infty}^{\infty} |h(\xi)|\,\mathrm{d}\xi < \infty
$$

\textbf{Rechnung:}

Da $h(\xi) \geq 0$ fĂĽr alle $\xi$ und $h(\xi) = 0$ fĂĽr $\xi < 0$ gilt:

$$
\int_{-\infty}^{\infty} |h(\xi)|\,\mathrm{d}\xi
= \int_0^{\infty} 3\, e^{-\xi/(4T)}\,\mathrm{d}\xi
= 3\left[-4T\, e^{-\xi/(4T)}\right]_0^{\infty}
= 3 \cdot 4T \cdot (0 - (-1)) = 12\,T.
$$

\textbf{LĂ¶sung:}

$$
\boxed{\;\int_{-\infty}^{\infty} |h(\xi)|\,\mathrm{d}\xi = 12\,T < \infty
\quad\Longrightarrow\quad \text{System ist BIBO-stabil.}\;}
$$

Die Impulsantwort ist fĂĽr jedes endliche $T > 0$ absolut integrierbar, das
System also BIBO-stabil. Anschaulich klingt die Exponentialfunktion schnell
genug ab, so dass ein beschrĂ¤nktes Eingangssignal stets ein beschrĂ¤nktes
Ausgangssignal erzeugt.

\end{beispiel}

% ---- b) --------------------------------------------------------
\subsubsection*{b) Faltung $y(t) = x(t) * h(t)$}

\textbf{Formel} (Faltungsintegral, Abschnitt~\ref{sec:faltungsintegral}):

$$
y(t) = \int_{-\infty}^{\infty} x(\tau)\, h(t - \tau)\,\mathrm{d}\tau
$$

\textbf{Rechnung:}

Der Integrand ist nur dort von null verschieden, wo \emph{beide} Faktoren
ungleich null sind. $x(\tau)$ verlangt $2T \leq \tau < 4T$, und
$h(t - \tau)$ verlangt $t - \tau \geq 0$, also $\tau \leq t$. Der Integrand
lautet dort $x(\tau)\,h(t-\tau) = 3 \cdot 3\, e^{-(t-\tau)/(4T)} = 9\, e^{-(t-\tau)/(4T)}$.
Es sind drei FĂ¤lle zu unterscheiden.

\emph{Fall 1 -- $t < 2T$ (noch keine Ăśberlappung):} Die Bedingung $\tau \leq t$
schlieĂźt das gesamte Rechteck aus, somit

$$
y(t) = 0.
$$

\emph{Fall 2 -- $2T \leq t < 4T$ (das System â€žlĂ¤dt sich auf``):}
Die Ăśberlappung reicht von $\tau = 2T$ bis $\tau = t$:

$$
y(t) = 9 \int_{2T}^{t} e^{-(t-\tau)/(4T)}\,\mathrm{d}\tau
     = 9 \left[4T\, e^{-(t-\tau)/(4T)}\right]_{\tau = 2T}^{\tau = t}
     = 36\,T\left(1 - e^{-(t - 2T)/(4T)}\right).
$$

\emph{Fall 3 -- $t \geq 4T$ (das gesamte Rechteck ist ĂĽberstrichen, â€žAbklingen``):}
Die Ăśberlappung umfasst das volle Rechteck von $\tau = 2T$ bis $\tau = 4T$:

$$
y(t) = 9 \int_{2T}^{4T} e^{-(t-\tau)/(4T)}\,\mathrm{d}\tau
     = 36\,T\left(e^{-(t - 4T)/(4T)} - e^{-(t - 2T)/(4T)}\right)
     = 36\,T\left(e - \sqrt{e}\,\right) e^{-t/(4T)}.
$$

Kontrolle der Stetigkeit an den Nahtstellen: Bei $t = 2T$ liefert Fall~2
$y(2T) = 0$, bei $t = 4T$ liefern Fall~2 und Fall~3 ĂĽbereinstimmend
$y(4T) = 36\,T\,(1 - e^{-1/2}) \approx 14{,}17\,T$. \checkmark

\textbf{LĂ¶sung:}

$$
\boxed{\;y(t) = \begin{cases}
0 & , \quad t < 2T \\[4pt]
36\,T\left(1 - e^{-(t - 2T)/(4T)}\right) & , \quad 2T \leq t < 4T \\[4pt]
36\,T\left(e^{-(t - 4T)/(4T)} - e^{-(t - 2T)/(4T)}\right) & , \quad t \geq 4T
\end{cases}\;}
$$

Das Ausgangssignal steigt im Intervall $[2T, 4T)$ von $0$ auf den Maximalwert
$36\,T\,(1 - e^{-1/2}) \approx 14{,}17\,T$ an (aufladender Ast) und klingt
danach exponentiell mit der Zeitkonstante $4T$ wieder ab. Die abrupten Flanken
des Rechtecks werden durch die Exponential-Impulsantwort â€žverschliffen`` -- ein
typisches Tiefpassverhalten.
