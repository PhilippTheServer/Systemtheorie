\section{Kombination und Permutation}

Es sind $n$ viele Sorten Eis vorhanden. 
Ein Hoernchen hat $k < n$ Kugeln Eis.

\textit{Wie viele Moeglichkeiten gibt es, das Hoernchen zu befuellen, wenn jede Sorte nur einmal vorkommen darf?}

\textbf{Question 1:} Die Reihenfolge der Kugeln ist wichtig.

$$
  n(n-1)(n-2) ... (n-(k-1)) = \frac{n!}{(n-k)!}
$$

\textbf{Question 2:} Die Reihenfolge der Kugeln ist nicht wichtig.

$$
  ABC = ACB = BCA = BAC = CBA = CAB
$$

Die Anzahl an Permutation durch $k$.

$$
  \frac{n!}{(n-k)!k!} = \binom{n}{k} = "n\ choose\ k"
$$

\textit{Wie viele Moeglichkeiten gibt es, das Hoernchen zu befuellen, wenn jede Sorte mehrfach genutzt werden darf?}

\textbf{Question 3:} Die Reihenfolge der Kugeln ist wichtig.

$$
  n \cdot n \cdot n ... n = n^k 
$$

\textbf{Question 4:} Die Reihenfolge der Kugeln ist nicht wichtig.

$$
  \binom{n - 1 + k}{k}
$$

\subsection{Ergebnisraum und Ereignisse}

\begin{definition}
  Wahrscheinlichkeit ist anhand bestimmter Experimenter definiert.

  $\Omega =$ Ereignisraum des Experimentes. Menge aller moeglichen Ergebnisse.

  \textbf{Definition}: Ein \textit{Ereignis}, $E$ ist eine Teilmenge des Ereignisraum $\Omega$.
\end{definition}

\end{definition}

\section{Diskrete und Kontinuierliche Ereignisse}

\begin{definition}
  Ein Ereignisraum ist Diskret wenn die Anzahl an Ereignissen endlich ist.

  Ein Ereignisraum ist Kontinuierlich wenn die Anzahl an Ereignissen unendlich ist.
\end{definition}

\textit{Welche der Experimente haben Diskrete/Kontinuierliche Ereignisraeume?}

\begin{enumerate}
  \item Einen Wuerfel 2 mal rollen
  \item Einen Dart in einem bestimmten intervall werfen
  \item Eine Muenze werfen bis das erste Mal Kopf kommt
  \item Die Zeit markerien wenn die 100. Email kommt
\end{enumerate}

$P\{E\}$ = Wahrscheinlichkeit des Ereignisses $E$.

$P\{E\}$ = Wahrscheinlichkeit, dass das Ergebnis des Experimentes in der Menge $E$ liegt.

\begin{definition}[Die 3 Wahrscheinlichkeits Axiome]
  \textbf{Nicht Negativitaet:} $P\{E\} \geq 0$ fuer alle Ereignisse $E$.\\
  \textbf{Additiviaet:} Wenn $E_1, E_2, ... E_n$ ein zaehlbare Menge an disjunkten Ereignissen ist, dann ist: $P\{E_1 \cup E_2 \cup E_3 \cup ...\} = P\{E_1\} + P\{E_2\} + P\{E_3\} ...$\\
  \textbf{Normalisierung:} $P\{\Omega\} = 1$
\end{definition}

\subsection{Ableitungen aus den 3 Wahrscheinlichkeits Axiomen}

\begin{Satz}
  $$
  P\{E \cup F\} = P\{E\} + P\{F\} - P\{E \cap F\}
  $$
\end{Satz}

\begin{Beweis}
  Wir druecken $E \cup F$ als eine Sammlung unabhängiger Mengen:\\
  $$
  E \cup F = E \cup (F| (E \cap F))
  $$

  Mittels der Additiviaet machen wir zwei Beobachtung:
  $$
    \begin{array}{rl}
      P\{E \cup F\} = & P\{E\} + P\{F| (E \cap F)\}\\
      P\{F\} = & P\{F| (E \cap F)\} + P\{E \cap F\}
    \end{array}
  $$
  Nun setze die zweite Gleichung in die erste ein.
\end{Beweis}

\begin{Satz}
$$
P\{E \cup F\} \leq P\{E\} + P\{F\}
$$
\end{Satz}

\textbf{Question:} Es wird ein Dart geworfen der ueberall gleich wahrscheinlich landen kann, zwischen $[0,1]$.
Was ist $P\{Dart\ landet\ auf\ 0.3\}$?

\textit{Hinweis:}: nutze die drei Axiome.

Die Wahrscheinlichkeit, dass der Dart genau auf $0.3$ landet ist $\frac{1}{n}$.\\
Wobei $n$ im Raum zwischen $[0,1]$ liegt.

\section{Bedingte Wahrscheinlichkeiten ueber Ereignisse}

\begin{Definition}
Die Bedingte Wahrscheinlichkeit das Ereignis $E$ bedingt durch Ereignis $F$ eintritt ist:\\
$$
P\{E|F\} = \frac{P\{E \cup F\}}{P\{F\}}
$$
Angenommen das $P\{F\} > 0$.
\end{Definition}

\section{Bedingte Wahrscheinlichkeit}

$$
P\{E_2|E_1\} = \frac{P\{E_1 \cap E_2\}}{P\{E_1\}}
$$

Wir koennen auch schreiben:
$$
P\{E_1 \cap E_2\} = P\{E_1\} \cdot P\{E_2|E_1\}
$$

Links ist die Menge aller Wahrscheinlichkeiten in beiden Mengen.

Die Mitte zeigt die Wahrscheinlichkeit, dass das Ergebnis in Menge $E_1$ liegt.

Rechts zeigt die Wahrscheinlichkeit, dass das Ergebnis in $E_2$ liegt, welches eine Teilmenge von $E_1$ ist.

\section{Kettenregel fuer Bedingungen}

\begin{Definition}
  $$
  P\{E_1 \cap E_2\} = P\{E_1\} \cdot P\{E_2|E_1\}
  $$
\end{Definition}

Das kann generallisiert dargestellt werden:

\begin{Satz}[Kettenregel fuer Bedingungen]
  $$
  P\{E_1 \cap E_2 \cap E_3 \cap ... \cap E_n\} = P\{E_1\} \cdot P\{E_2|E_1\} \cdot P\{E_3|E_1 \cap E_2\} ... P\{E_n| E_1 \cap E_2 \cap ... \cap E_{n-1}\}
  $$
\end{Satz}
